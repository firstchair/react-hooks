# useDocumentEventListener

This hook allows you to add a document event listener and remove it when the component unmounts.

## Reference

```ts
export function useDocumentEventListener<T extends keyof DocumentEventMap>(
  event: T,
  callback: (event: DocumentEventMap[T]) => void,
): void;
```

## Usage

```tsx
const [keydown, setKeydown] = useState<ReadonlyArray<string>>([]);

useDocumentEventListener('keydown', (event) => {
  setKeydown((previous) => [...previous, event.key]);
});
```
